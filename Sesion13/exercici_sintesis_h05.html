Exercicis de síntesis - CNfull9.pdf

clearvars;

PVI

f=@(t,y)t.*exp(3*t)-2*y; % y'=f(t,y)
a = 0; alpha = 0; % y(a)=alpha
b = 2.0; % y(b)=??
h = 0.05; %h=0.25,0.1 % pas h
t=[a:h:b];

Solució analítica

g=@(t)((5.*t-1).*exp(3*t)+exp(-2*t))/25;
yexa=g(t); y=abs(yexa);

Mètodes de resolució aproximada

fprintf('resultats per h = %0.5f',h)
resultats per h = 0.05000

Euler

[ ye ] = Euler( f,a,b,h,alpha );
errorlocal_euler = abs(yexa-ye); %% error local
error_euler = norm(yexa-ye,'inf') %% error màxim
error_euler = 8.2487

Euler millorat

[ yml ] = EulerMill(f,a,b,h,alpha);
errorlocal_heun = abs(yexa-yml); %% error local
error_mill = norm(yexa-yml,'inf') %% error màxim
error_mill = 0.66734

Euler modificat

[ ym ] = EulerMod(f,a,b,h,alpha);
errorlocal_mod = abs(yexa-ym); %% error local
error_mod = norm(yexa-ym,'inf') %% error màxim
error_mod = 0.079977

RK4

[ yk ] = RK4(f,a,b,h,alpha);
errorlocal_rk4 = abs(yexa-yk); %% error local
error_rk4 = norm(yexa-yk,'inf') %% error màxim
error_rk4 = 0.00038347

Gràfics

clf
subplot(211)
plot(t,yexa,'r',t,ye,'b',t,yml,'cyan',t,ym,'m',t,yk,'g')
title('y''=te^{3*t}-2*y')
legend('Y','EU','EML','EMD','RK4','Location','bestoutside')
subplot(212)
plot(t,fix(-log10(errorlocal_euler./y)),'*b'),hold on
plot(t,fix(-log10(errorlocal_heun./y)),'+cyan')
plot(t,fix(-log10(errorlocal_mod./y)),'sm')
plot(t,fix(-log10(errorlocal_rk4./y)),'dg'),title('xifres significatives')
legend('E','EML','EMD','RK4','Location','bestoutside')
axis([a,b,0,6])
